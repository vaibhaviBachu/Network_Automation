Code Examples

Basic Serialization in Go
package main

import (
    "fmt"
    "github.com/saichler/l8srlz/go/serialize/object"
)

func main() {
    // Serialize any Go object
    data := map[string]interface{}{
        "name": "John Doe",
        "age":  30,
        "active": true,
    }
    
    // Serialize to binary
    serialized, err := object.DataOf(data)
    if err != nil {
        panic(err)
    }
    
    fmt.Printf("Serialized %d bytes\n", len(serialized))
    
    // Deserialize back
    deserialized, err := object.ElemOf(serialized, registry)
    if err != nil {
        panic(err)
    }
    
    fmt.Printf("Deserialized: %+v\n", deserialized)
}
Working with Elements
// Create elements container
elements := object.New(nil, myData)

// Add multiple items
elements.Add("item1", "key1", nil)
elements.Add("item2", "key2", nil)

// Serialize all elements
data, err := elements.Serialize()
if err != nil {
    return err
}

// Deserialize elements
newElements := &object.Elements{}
err = newElements.Deserialize(data, registry)
if err != nil {
    return err
}
Java Implementation
import com.saichler.l8srlz.serialize.object.*;

public class SerializationExample {
    public static void main(String[] args) {
        // Create serialization object
        SerializationObject obj = new SerializationObject();
        
        // Add data to serialize
        obj.add("Hello World");
        obj.add(42);
        obj.add(true);
        
        // Serialize to byte array
        byte[] data = obj.toByteArray();
        
        // Create deserializer
        SerializationObject deserializer = new SerializationObject(data);
        
        // Read back the data
        String text = (String) deserializer.get();
        Integer number = (Integer) deserializer.get();
        Boolean flag = (Boolean) deserializer.get();
        
        System.out.println("Text: " + text);
        System.out.println("Number: " + number);
        System.out.println("Flag: " + flag);
    }
}
Protocol Buffers Integration
// Define your protobuf message
message := &pb.UserProfile{
    Name:  "Alice",
    Email: "alice@example.com",
    Age:   25,
}

// Register the type
registry.Register("UserProfile", &pb.UserProfile{})

// Serialize (automatically detects protobuf)
data, err := object.DataOf(message)
if err != nil {
    return err
}

// Deserialize with type resolution
result, err := object.ElemOf(data, registry)
if err != nil {
    return err
}

profile := result.(*pb.UserProfile)
fmt.Printf("Name: %s, Email: %s\n", profile.Name, profile.Email)
Replica Support Example
// Create original elements container
original := object.New(nil, myData)

// Add some data
original.Add("item1", "key1", nil)
original.Add("item2", "key2", nil)

// Create replica request with replica number
replicaRequest := object.NewReplicaRequest(original, byte(1))

// Check if element is a replica
if replicaRequest.IsReplica() {
    replicaNum := replicaRequest.Replica()
    fmt.Printf("Processing replica #%d\n", replicaNum)
}

// Serialize replica request for transmission
data, err := replicaRequest.Serialize()
if err != nil {
    return err
}

// Send to replica nodes...
fmt.Printf("Replica data size: %d bytes\n", len(data))

// On the receiving end
received := &object.Elements{}
err = received.Deserialize(data, registry)
if err != nil {
    return err
}

// Check if received data is a replica
if received.IsReplica() {
    fmt.Printf("Received replica #%d\n", received.Replica())
    // Process replica data...
}
