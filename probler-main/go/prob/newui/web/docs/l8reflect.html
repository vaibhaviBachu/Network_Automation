Layer 8 Model Agnostic Infra

Advanced reflection capabilities for deep cloning, introspection, property management, and change tracking in Go applications

Get Started View Examples

Powerful Features

Advanced Introspection

Sophisticated struct analysis and type registration with node-based representation, caching, and table view generation for complex nested types.

Deep Cloning

Type-safe deep cloning of any Go data structure with circular reference handling, support for all Go types, and customizable field filtering.

Property Management

Property-based access to struct fields with hierarchical paths, key support, and getter/setter abstractions for different data types.

Change Tracking

Intelligent diff detection between object versions with granular tracking, update application, validation, and rollback support.

Code Examples

Basic Introspection 
import (
    "github.com/saichler/l8reflect/go/reflect/introspecting"
    "github.com/saichler/l8types/go/ifs"
)

// Create an introspector
registry := // your registry implementation
introspector := introspecting.NewIntrospect(registry)

// Inspect a struct
type Person struct {
    Name string
    Age  int
    Address *Address
}

person := &Person{Name: "John", Age: 30}
node, err := introspector.Inspect(person)
if err != nil {
    log.Fatal(err)
}

// Access node information
fmt.Printf("Type: %s\n", node.TypeName)
fmt.Printf("Attributes: %d\n", len(node.Attributes))

// Get table view
tableView, exists := introspector.TableView("Person")
if exists {
    fmt.Printf("Columns: %d\n", len(tableView.Columns))
}
Deep Cloning 
import "github.com/saichler/l8reflect/go/reflect/cloning"

// Create a cloner
cloner := cloning.NewCloner()

// Clone any object
original := &Person{
    Name: "Alice", 
    Age: 25,
    Address: &Address{Street: "123 Main St"},
}

// Deep clone with circular reference handling
cloned := cloner.Clone(original).(*Person)

// Modify clone without affecting original
cloned.Name = "Bob"
cloned.Address.Street = "456 Oak Ave"

fmt.Printf("Original: %s at %s\n", 
    original.Name, original.Address.Street)
fmt.Printf("Clone: %s at %s\n", 
    cloned.Name, cloned.Address.Street)

// Fields automatically skipped during cloning:
// - "DoNotCompare", "DoNotCopy"
// - Fields starting with "XXX"
// - Private fields (lowercase first letter)
Property Access 
import (
    "github.com/saichler/l8reflect/go/reflect/properties"
    "github.com/saichler/l8types/go/types"
)

// Create a property from a node and resources
prop := properties.NewProperty(node, nil, key, value, resources)

// Get property value
value := prop.Value()

// Check property type
if prop.IsString() {
    fmt.Println("Property is a string")
}

// Check if property is a leaf node
if prop.IsLeaf() {
    fmt.Println("This is a leaf property")
}

// Navigate complex object hierarchies using property paths
// Example: "person.address.street"
property, err := properties.PropertyOf("person.address<123>.street", resources)
if err != nil {
    log.Fatal(err)
}

// Get property ID for tracking
propertyId, err := property.PropertyId()
if err != nil {
    log.Fatal(err)
}
fmt.Printf("Property ID: %s\n", propertyId)
Change Tracking & Updates 
import "github.com/saichler/l8reflect/go/reflect/updating"

// Create an updater
// Parameters: resources, isNilValid, newItemIsFull
updater := updating.NewUpdater(resources, false, true)

// Track changes between two versions
oldPerson := &Person{Name: "John", Age: 30, City: "NYC"}
newPerson := &Person{Name: "John", Age: 31, City: "Boston"}

err := updater.Update(oldPerson, newPerson)
if err != nil {
    log.Fatal(err)
}

// Get the changes with granular tracking
changes := updater.Changes()
for _, change := range changes {
    fmt.Printf("Property: %s\n", change.Property().PropertyId())
    fmt.Printf("Changed: %v -> %v\n", change.OldValue, change.NewValue)
    fmt.Printf("Change Type: %s\n", change.Type())
}

// Changes provide detailed information:
// - Property path and context
// - Old and new values
// - Change type (add, update, delete)
// - Validation and rollback support
Advanced Decorators 
import "github.com/saichler/l8reflect/go/reflect/introspecting"

// ========== AlwaysOverwrite Decorator ==========
// Force complete map structure updates during change tracking
node, _ := introspector.Inspect(&MyMapStruct{})
introspecting.AddAlwayOverwriteDecorator(node)

// When this decorator is present, the entire map will be replaced
// during updates instead of merging individual keys
// Useful for configuration maps where you want full replacement

// ========== Multiple Primary Keys ==========
// Define composite primary keys with multiple attributes
type ServiceIndex struct {
    ServiceName string
    ServiceArea byte
    Data        map[string]interface{}
}

node, _ := introspector.Inspect(&ServiceIndex{})
introspecting.AddPrimaryKeyDecorator(node, "ServiceName", "ServiceArea")

// The updater will use both fields as a composite key
// when tracking changes and identifying unique instances
updater := updating.NewUpdater(resources, false, false)
err := updater.Update(oldService, newService)

// ========== NoNestedInspection Decorator ==========
// Prevent deep inspection of certain fields
node, _ := introspector.Inspect(&ComplexStruct{})
introspecting.AddNoNestedInspection(node)

// This decorator skips nested inspection for performance
// when you don't need deep analysis of certain structures

// ========== Field Filtering ==========
// Automatic field filtering during operations
type MyStruct struct {
    Name         string      // Included
    DoNotCompare string      // Skipped automatically
    DoNotCopy    interface{} // Skipped automatically
    XXXDebug     string      // Skipped (starts with XXX)
    privateField string      // Skipped (lowercase first letter)
}

Introspector API

NewIntrospect(registry ifs.IRegistry) *Introspector

Creates a new introspector with the provided registry.

introspector := introspecting.NewIntrospect(registry)

Inspect(any interface{}) (*types.RNode, error)

Inspects a struct and returns its node representation.

node, err := introspector.Inspect(structInstance)

Node(path string) (*types.RNode, bool)

Retrieves a cached node by path.

node, exists := introspector.Node("person")

TableView(name string) (*types.TableView, bool)

Gets a table view for the specified type.

tableView, exists := introspector.TableView("Person")

Decorators API

AddAlwayOverwriteDecorator(node *l8reflect.L8Node)

Forces full replacement of map structures during updates instead of merging.

introspecting.AddAlwayOverwriteDecorator(node)

AddPrimaryKeyDecorator(node *l8reflect.L8Node, fields ...string)

Defines multiple fields as composite primary keys for unique identification.

introspecting.AddPrimaryKeyDecorator(node, "ServiceName", "ServiceArea")

AddNoNestedInspection(node *l8reflect.L8Node)

Prevents deep inspection of certain fields for performance optimization.

introspecting.AddNoNestedInspection(node)

AlwaysFullDecorator(node *l8reflect.L8Node) bool

Checks if the AlwaysFull decorator is set on a node.

if introspecting.AlwaysFullDecorator(node) { // Handle full replacement }

Cloner API

NewCloner() *Cloner

Creates a new cloner instance.

cloner := cloning.NewCloner()

Clone(any interface{}) interface{}

Performs deep cloning of any Go data structure.

cloned := cloner.Clone(original)
